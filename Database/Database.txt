);

CREATE TABLE customers(
  c_name varchar2(255),
  c_creditcard_num varchar2(255),
  c_email varchar2(255),
  c_password varchar2(255),
  c_phone varchar2(255),
  c_address varchar2(255),
  PRIMARY KEY(c_email)
);


CREATE TABLE admins(
  email varchar(255)NOT NULL,
  adminPassword varchar(254),
  PRIMARY KEY(email)
);

CREATE TABLE drivers(
  d_name varchar(255),
  d_creditcard_num varchar(255),
  d_email varchar(255),
  d_password varchar(254),
  d_phone varchar(254),
  d_address varchar(255),
  PRIMARY KEY(d_email)
);
